# DOM MANIPULATION (TO ACCESS ELEMENT)
SELECT WITH ID
document.getElementBYId("myId")  (when we search for an id that does not exist that it will give null)

SELECT WITH CLASS
document.getElementBYClassName("myclass")  (when we search for a class that does not exist than we get an empty html object)


# SELECTING WITH TAG
document.getElementbyTagName("p")

# QUERY SELECTOR
document.querySelector("id/class/element")  // to SELECT first element if you wan to SELECT all element use 
document.querySelectorAll("id/class/tag")  use .for id and # for class 

#### properties  (get and set)
tagName : returns the tag name of the node
innerText : returns the text content of the element and all its children
innerHTML : returns the plain text or HTML contents in the elements
textContent : return textual content even for hidden elements
